API contract (from backend/ecpm/modeling/schemas.py):
```
GET /api/forecasting/forecasts -> { forecasts: { [indicator]: { indicator, horizon_quarters, forecasts: [{date, point, lower_68, upper_68, lower_95, upper_95}], model_info } }, generated_at }
GET /api/forecasting/regime -> { current_regime, regime_label, regime_probabilities, transition_matrix, smoothed_probabilities, regime_durations }
GET /api/forecasting/crisis-index -> { current_value, trpf_component, realization_component, financial_component, history: [{date, composite, trpf, realization, financial}] }
GET /api/forecasting/backtests -> { backtests: [{ episode_name, start_date, end_date, crisis_index_series, warning_12m, warning_24m, peak_value, peak_date }], generated_at }
POST /api/forecasting/train -> { task_id, status }
GET /api/forecasting/training/stream -> SSE events: connected, progress({step, status, detail}), heartbeat
GET /api/forecasting/training/status -> { task_id, status, current_step, steps, started_at, completed_at, error }
```
